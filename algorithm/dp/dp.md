斐波那切数列

int fib(int n){
    vector<int> mem(n+1, -1);
    mem[0] = 0; 
    mem[1] = 1;
    for(int i = 2; i < n; i++)
        mem[n] = mem[n - 1] + mem[n - 2];
    
    return mem[n];
}
